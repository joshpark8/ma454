\section{Soluble Groups II}

\begin{ttheorem}
  \begin{enumerate}
    \item If \( G \) is a soluble group with \( A\sgp G \), then \( A \) is soluble.
    \item Let \( H\pnsgp G \). Then \( G \) is soluble \iff~\( H\tand \qg{G}{H} \) are soluble.
  \end{enumerate}
\end{ttheorem}
